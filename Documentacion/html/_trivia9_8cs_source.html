215 void DibujarLinea(Vector3 inicio, Vector3 fin)
216 {
217 Vector3 localInicio = imagenLinea.rectTransform.parent.InverseTransformPoint(inicio);
218 Vector3 localFin = imagenLinea.rectTransform.parent.InverseTransformPoint(fin);
219
220 Vector3 medio = (localInicio + localFin) / 2f;
221 imagenLinea.rectTransform.localPosition = medio;
222
223 float distancia = Vector3.Distance(localInicio, localFin);
224 imagenLinea.rectTransform.sizeDelta = new Vector2(distancia, 5f); // 5f es el grosor de la línea
225
226 float angulo = Mathf.Atan2(localFin.y - localInicio.y, localFin.x - localInicio.x) * Mathf.Rad2Deg;
227 imagenLinea.rectTransform.localRotation = Quaternion.Euler(0, 0, angulo);
228 }
